// Enhanced wallet connection service
export class WalletConnectionService {
  constructor() {
    this.walletConnect = null
    this.connectionState = 'disconnected'
    this.walletType = null
    this.retryAttempts = 0
    this.maxRetries = 3
  }

  async initializeWalletConnect() {
    try {
      // Initialize WalletConnect v2
      this.walletConnect = await WalletConnect.init({
        projectId: process.env.VUE_APP_WALLETCONNECT_PROJECT_ID,
        relayUrl: 'wss://relay.walletconnect.com',
        metadata: {
          name: 'BCH HodlVault',
          description: 'Bitcoin Cash vault application',
          url: window.location.origin,
          icons: [`${window.location.origin}/favicon.ico`]
        }
      })

      // Set up event listeners
      this.setupEventListeners()
      
      console.log('WalletConnect initialized successfully')
      return true
    } catch (error) {
      console.error('WalletConnect initialization failed:', error)
      return false
    }
  }

  setupEventListeners() {
    if (!this.walletConnect) return

    this.walletConnect.on('session_proposal', this.onSessionProposal.bind(this))
    this.walletConnect.on('session_request', this.onSessionRequest.bind(this))
    this.walletConnect.on('session_delete', this.onSessionDelete.bind(this))
  }

  async connectWallet() {
    try {
      if (this.connectionState === 'connected') {
        console.log('Wallet already connected')
        return this.getSessionInfo()
      }

      // Detect wallet type
      this.walletType = this.detectWalletType()
      
      // Create session proposal
      const { uri, approval } = await this.walletConnect.pair({
        uri: this.generatePairingUri()
      })

      if (uri) {
        // Show QR code or deep link
        this.showConnectionQR(uri)
      }

      // Wait for session approval
      const session = await approval()
      
      this.connectionState = 'connected'
      this.retryAttempts = 0
      
      console.log('Wallet connected successfully')
      return this.getSessionInfo()
      
    } catch (error) {
      console.error('Wallet connection failed:', error)
      this.connectionState = 'failed'
      
      // Retry logic
      if (this.retryAttempts < this.maxRetries) {
        this.retryAttempts++
        console.log(`Retrying connection (attempt ${this.retryAttempts}/${this.maxRetries})`)
        await this.delay(1000 * this.retryAttempts)
        return this.connectWallet()
      }
      
      throw error
    }
  }

  detectWalletType() {
    // Check for Paytaca wallet
    if (window.paytaca) {
      return 'paytaca'
    }
    
    // Check for other BCH wallets
    if (window.bitcoinCashWallet) {
      return 'bitcoin-cash-wallet'
    }
    
    // Default to WalletConnect
    return 'walletconnect'
  }

  async signTransaction(transactionHex, retryCount = 0) {
    try {
      if (this.connectionState !== 'connected') {
        throw new Error('Wallet not connected')
      }

      // Validate transaction format
      this.validateTransactionFormat(transactionHex)

      let result
      
      // Use wallet-specific signing method
      switch (this.walletType) {
        case 'paytaca':
          result = await this.signWithPaytaca(transactionHex)
          break
        case 'walletconnect':
          result = await this.signWithWalletConnect(transactionHex)
          break
        default:
          throw new Error(`Unsupported wallet type: ${this.walletType}`)
      }

      // Verify signature result
      if (!result || !result.signature) {
        throw new Error('Invalid signature result from wallet')
      }

      console.log('Transaction signed successfully')
      return result
      
    } catch (error) {
      console.error('Transaction signing failed:', error)
      
      // Retry logic
      if (retryCount < this.maxRetries) {
        console.log(`Retrying transaction signing (attempt ${retryCount + 1}/${this.maxRetries})`)
        await this.delay(1000 * (retryCount + 1))
        return this.signTransaction(transactionHex, retryCount + 1)
      }
      
      throw error
    }
  }

  async signWithPaytaca(transactionHex) {
    try {
      // Paytaca specific signing method
      if (!window.paytaca) {
        throw new Error('Paytaca wallet not detected')
      }

      const result = await window.paytaca.signTransaction({
        transactionHex: transactionHex,
        network: 'chipnet'
      })

      return {
        signature: result.signature,
        transactionId: result.txid,
        rawTransaction: result.signedTx
      }
    } catch (error) {
      console.error('Paytaca signing failed:', error)
      throw new Error(`Paytaca signing failed: ${error.message}`)
    }
  }

  async signWithWalletConnect(transactionHex) {
    try {
      if (!this.walletConnect) {
        throw new Error('WalletConnect not initialized')
      }

      const session = this.walletConnect.session
      if (!session) {
        throw new Error('No active session')
      }

      // Get the first available account
      const accounts = Object.keys(session.namespaces)
      if (accounts.length === 0) {
        throw new Error('No accounts available')
      }

      const account = session.namespaces[accounts[0]].accounts[0]
      
      // Sign transaction using WalletConnect
      const result = await this.walletConnect.request({
        topic: session.topic,
        chainId: 'bch:chipnet',
        request: {
          method: 'bch_signTransaction',
          params: {
            transactionHex: transactionHex,
            address: account.split(':')[2]
          }
        }
      })

      return {
        signature: result.signature,
        transactionId: result.txid,
        rawTransaction: result.signedTransaction
      }
    } catch (error) {
      console.error('WalletConnect signing failed:', error)
      throw new Error(`WalletConnect signing failed: ${error.message}`)
    }
  }

  validateTransactionFormat(transactionHex) {
    if (!transactionHex || typeof transactionHex !== 'string') {
      throw new Error('Invalid transaction format: must be a hex string')
    }

    // Check if valid hex
    const hexRegex = /^[0-9a-fA-F]+$/
    if (!hexRegex.test(transactionHex)) {
      throw new Error('Invalid transaction format: not valid hex')
    }

    // Basic structure validation (simplified)
    try {
      // This would normally use a BCH library to validate
      if (transactionHex.length < 10) {
        throw new Error('Transaction too short')
      }
      
      if (transactionHex.length > 100000) {
        throw new Error('Transaction too long')
      }
    } catch (error) {
      throw new Error(`Transaction validation failed: ${error.message}`)
    }
  }

  async checkConnectionHealth() {
    try {
      if (this.connectionState !== 'connected') {
        return { healthy: false, reason: 'Not connected' }
      }

      // Ping the wallet to check connection
      const session = this.walletConnect?.session
      if (!session) {
        this.connectionState = 'disconnected'
        return { healthy: false, reason: 'No active session' }
      }

      // Try a simple request to test connection
      await this.walletConnect.request({
        topic: session.topic,
        chainId: 'bch:chipnet',
        request: {
          method: 'bch_getAccounts',
          params: {}
        }
      })

      return { healthy: true }
    } catch (error) {
      console.error('Connection health check failed:', error)
      this.connectionState = 'unhealthy'
      return { healthy: false, reason: error.message }
    }
  }

  getSessionInfo() {
    if (!this.walletConnect?.session) {
      return null
    }

    const session = this.walletConnect.session
    const accounts = Object.values(session.namespaces).flatMap(namespace => namespace.accounts)
    
    return {
      topic: session.topic,
      accounts: accounts,
      walletType: this.walletType,
      connectionState: this.connectionState
    }
  }

  showConnectionQR(uri) {
    // This would integrate with a QR code library
    console.log('Connection QR code URI:', uri)
    
    // Emit event for UI to handle QR code display
    if (typeof window !== 'undefined') {
      window.dispatchEvent(new CustomEvent('walletconnect:qr', { detail: { uri } }))
    }
  }

  generatePairingUri() {
    // Generate a random pairing URI for testing
    return `wc:${Math.random().toString(36).substr(2, 9)}@2?relay-protocol=irn&symKey=${Math.random().toString(36).substr(2, 32)}`
  }

  delay(ms) {
    return new Promise(resolve => setTimeout(resolve, ms))
  }

  async disconnect() {
    try {
      if (this.walletConnect?.session) {
        await this.walletConnect.disconnect({
          topic: this.walletConnect.session.topic,
          reason: {
            code: 6000,
            message: 'User disconnected'
          }
        })
      }
      
      this.connectionState = 'disconnected'
      this.walletType = null
      this.retryAttempts = 0
      
      console.log('Wallet disconnected successfully')
    } catch (error) {
      console.error('Wallet disconnection failed:', error)
    }
  }
}

console.log('Connected to wallet:', sessionInfo)

console.log('Connection health:', healthStatus)
