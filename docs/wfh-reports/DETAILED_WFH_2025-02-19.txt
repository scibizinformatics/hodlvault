<script>
import { defineComponent } from 'vue'
import { vaultStorage } from 'src/services/vault-storage'
import { fetchOraclePrice } from 'src/services/oracle'

export default defineComponent({
  name: 'MyVaultsPage',

  data() {
    return {
      vaults: [],
      loading: false,
      searchQuery: '',
      filterStatus: null,
      selectedVaultId: null,
      currentBchPrice: null,
      vaultMenu: {
        show: false,
        vault: null,
        target: null
      },
      filterOptions: [
        { label: 'All Vaults', value: null },
        { label: 'Active', value: 'active' },
        { label: 'Completed', value: 'completed' },
        { label: 'Pending', value: 'pending' }
      ]
    }
  },

  computed: {
    filteredVaults() {
      let filtered = this.vaults

      // Apply search filter
      if (this.searchQuery) {
        const query = this.searchQuery.toLowerCase()
        filtered = filtered.filter(vault => 
          vault.name?.toLowerCase().includes(query) ||
          vault.contractAddress.toLowerCase().includes(query) ||
          vault.priceTarget.toString().includes(query)
        )
      }

      // Apply status filter
      if (this.filterStatus) {
        filtered = filtered.filter(vault => vault.status === this.filterStatus)
      }

      return filtered
    }
  },

  methods: {
    async loadVaults() {
      this.loading = true
      try {
        // Load vaults from storage
        this.vaults = vaultStorage.getAllVaults()
        
        // Sort by creation date (newest first)
        this.vaults.sort((a, b) => (b.createdAt || 0) - (a.createdAt || 0))
        
        // Get current BCH price for USD calculations
        await this.loadCurrentPrice()
        
      } catch (error) {
        console.error('Failed to load vaults:', error)
        this.$q.notify({
          type: 'negative',
          message: 'Failed to load vaults'
        })
      } finally {
        this.loading = false
      }
    },

    async loadCurrentPrice() {
      try {
        const priceData = await fetchOraclePrice()
        this.currentBchPrice = priceData.price
      } catch (error) {
        console.error('Failed to load current price:', error)
        this.currentBchPrice = 350 // Fallback price
      }
    },

    selectVault(vault) {
      this.selectedVaultId = vault.id
      // Store selected vault for navigation to vault page
      localStorage.setItem('hodl-vault-selected-vault', JSON.stringify(vault))
      
      // Navigate to vault page
      this.$router.push('/vault')
    },

    async refreshVault(vault) {
      vault.refreshing = true
      try {
        // Refresh vault balance
        const newBalance = await this.getVaultBalance(vault.contractAddress)
        vault.balance = newBalance
        
        // Update vault in storage
        vaultStorage.updateVault(vault.id, { balance: newBalance })
        
        this.$q.notify({
          type: 'positive',
          message: 'Vault refreshed successfully'
        })
      } catch (error) {
        console.error('Failed to refresh vault:', error)
        this.$q.notify({
          type: 'negative',
          message: 'Failed to refresh vault'
        })
      } finally {
        vault.refreshing = false
      }
    },

    async getVaultBalance(contractAddress) {
      // This would call the blockchain service to get balance
      // For now, return a mock value
      return 0.05 // 0.05 BCH
    },

    getStatusColor(status) {
      const colors = {
        active: 'positive',
        completed: 'info',
        pending: 'warning',
        failed: 'negative'
      }
      return colors[status] || 'grey'
    },

    getStatusLabel(status) {
      const labels = {
        active: 'Active',
        completed: 'Completed',
        pending: 'Pending',
        failed: 'Failed'
      }
      return labels[status] || 'Unknown'
    },

    formatBalance(balance) {
      return balance ? balance.toFixed(8) : '0.00000000'
    },

    calculateUSDValue(balance, price) {
      if (!balance || !price) return 0
      return (balance * price).toFixed(2)
    },

    getProgressPercentage(vault) {
      if (!this.currentBchPrice || !vault.priceTarget) return 0
      const percentage = (this.currentBchPrice / vault.priceTarget) * 100
      return Math.min(Math.round(percentage), 100)
    },

    showVaultMenu(vault, event) {
      this.vaultMenu = {
        show: true,
        vault: vault,
        target: event.target
      }
    },

    viewVaultDetails(vault) {
      // Navigate to vault details page
      this.selectVault(vault)
    },

    editVault(vault) {
      // Open edit dialog
      this.$q.dialog({
        title: 'Edit Vault',
        message: 'Edit vault name:',
        prompt: {
          model: vault.name || '',
          type: 'text'
        },
        cancel: true,
        persistent: true
      }).onOk(async (newName) => {
        try {
          await vaultStorage.updateVault(vault.id, { name: newName })
          vault.name = newName
          this.$q.notify({
            type: 'positive',
            message: 'Vault updated successfully'
          })
        } catch (error) {
          console.error('Failed to update vault:', error)
          this.$q.notify({
            type: 'negative',
            message: 'Failed to update vault'
          })
        }
      })
    },

    duplicateVault(vault) {
      // Create new vault with same settings
      this.$router.push({
        path: '/vault',
        query: {
          duplicate: vault.id
        }
      })
    },

    exportVault(vault) {
      const data = JSON.stringify(vault, null, 2)
      const blob = new Blob([data], { type: 'application/json' })
      const url = URL.createObjectURL(blob)
      
      const a = document.createElement('a')
      a.href = url
      a.download = `vault-${vault.id}.json`
      a.click()
      
      URL.revokeObjectURL(url)
      
      this.$q.notify({
        type: 'positive',
        message: 'Vault data exported successfully'
      })
    },

    confirmDeleteVault(vault) {
      this.$q.dialog({
        title: 'Delete Vault',
        message: `Are you sure you want to delete "${vault.name || vault.id}"? This action cannot be undone.`,
        cancel: true,
        persistent: true,
        ok: {
          color: 'negative',
          label: 'Delete'
        }
      }).onOk(async () => {
        try {
          await vaultStorage.deleteVault(vault.id)
          this.vaults = this.vaults.filter(v => v.id !== vault.id)
          this.$q.notify({
            type: 'positive',
            message: 'Vault deleted successfully'
          })
        } catch (error) {
          console.error('Failed to delete vault:', error)
          this.$q.notify({
            type: 'negative',
            message: 'Failed to delete vault'
          })
        }
      })
    }
  },

  mounted() {
    this.loadVaults()
  }
})
</script>

// Enhanced vault storage service
export class VaultStorage {
  constructor() {
    this.storageKey = 'hodl-vault-data'
    this.version = '1.0'
  }

  getAllVaults() {
    try {
      const data = localStorage.getItem(this.storageKey)
      if (!data) return []

      const parsed = JSON.parse(data)
      
      // Validate data structure
      if (!Array.isArray(parsed.vaults)) {
        console.warn('Invalid vault data structure, returning empty array')
        return []
      }

      return parsed.vaults.map(vault => this.validateVaultData(vault))
    } catch (error) {
      console.error('Failed to load vaults:', error)
      return []
    }
  }

  saveVault(vaultData) {
    try {
      const existingData = this.getStorageData()
      
      // Validate vault data
      const validatedVault = this.validateVaultData(vaultData)
      
      // Add metadata
      validatedVault.createdAt = validatedVault.createdAt || Date.now()
      validatedVault.updatedAt = Date.now()
      validatedVault.version = this.version
      
      // Check for duplicates
      const existingIndex = existingData.vaults.findIndex(v => v.id === validatedVault.id)
      
      if (existingIndex >= 0) {
        // Update existing vault
        existingData.vaults[existingIndex] = validatedVault
      } else {
        // Add new vault
        existingData.vaults.push(validatedVault)
      }
      
      // Save to storage
      localStorage.setItem(this.storageKey, JSON.stringify(existingData))
      
      return validatedVault
    } catch (error) {
      console.error('Failed to save vault:', error)
      throw new Error('Failed to save vault data')
    }
  }

  updateVault(vaultId, updates) {
    try {
      const vaults = this.getAllVaults()
      const vaultIndex = vaults.findIndex(v => v.id === vaultId)
      
      if (vaultIndex === -1) {
        throw new Error('Vault not found')
      }
      
      // Update vault
      vaults[vaultIndex] = {
        ...vaults[vaultIndex],
        ...updates,
        updatedAt: Date.now()
      }
      
      // Save updated data
      const data = {
        version: this.version,
        vaults: vaults,
        lastUpdated: Date.now()
      }
      
      localStorage.setItem(this.storageKey, JSON.stringify(data))
      
      return vaults[vaultIndex]
    } catch (error) {
      console.error('Failed to update vault:', error)
      throw new Error('Failed to update vault')
    }
  }

  deleteVault(vaultId) {
    try {
      const data = this.getStorageData()
      const vaultIndex = data.vaults.findIndex(v => v.id === vaultId)
      
      if (vaultIndex === -1) {
        throw new Error('Vault not found')
      }
      
      // Remove vault
      data.vaults.splice(vaultIndex, 1)
      data.lastUpdated = Date.now()
      
      // Save updated data
      localStorage.setItem(this.storageKey, JSON.stringify(data))
      
      return true
    } catch (error) {
      console.error('Failed to delete vault:', error)
      throw new Error('Failed to delete vault')
    }
  }

  validateVaultData(vault) {
    // Required fields
    const requiredFields = ['id', 'contractAddress', 'priceTarget', 'ownerPkhHex']
    
    for (const field of requiredFields) {
      if (!vault[field]) {
        throw new Error(`Missing required field: ${field}`)
      }
    }

    // Validate data types
    if (typeof vault.priceTarget !== 'number' || vault.priceTarget <= 0) {
      throw new Error('Invalid price target')
    }

    if (typeof vault.contractAddress !== 'string' || !vault.contractAddress.startsWith('bitcoincash:')) {
      throw new Error('Invalid contract address')
    }

    // Set defaults for optional fields
    return {
      ...vault,
      name: vault.name || null,
      balance: vault.balance || 0,
      status: vault.status || 'active',
      createdAt: vault.createdAt || Date.now(),
      updatedAt: vault.updatedAt || Date.now()
    }
  }

  getStorageData() {
    try {
      const data = localStorage.getItem(this.storageKey)
      return data ? JSON.parse(data) : { version: this.version, vaults: [], lastUpdated: Date.now() }
    } catch (error) {
      console.error('Failed to parse storage data:', error)
      return { version: this.version, vaults: [], lastUpdated: Date.now() }
    }
  }

  exportVaults() {
    const vaults = this.getAllVaults()
    const exportData = {
      version: this.version,
      exportDate: new Date().toISOString(),
      vaults: vaults
    }
    
    return JSON.stringify(exportData, null, 2)
  }

  importVaults(importData) {
    try {
      const data = typeof importData === 'string' ? JSON.parse(importData) : importData
      
      if (!Array.isArray(data.vaults)) {
        throw new Error('Invalid import data format')
      }
      
      // Validate each vault
      const validatedVaults = data.vaults.map(vault => this.validateVaultData(vault))
      
      // Merge with existing vaults (avoid duplicates)
      const existingVaults = this.getAllVaults()
      const existingIds = new Set(existingVaults.map(v => v.id))
      
      const newVaults = validatedVaults.filter(v => !existingIds.has(v.id))
      
      const mergedVaults = [...existingVaults, ...newVaults]
      
      // Save merged data
      const storageData = {
        version: this.version,
        vaults: mergedVaults,
        lastUpdated: Date.now()
      }
      
      localStorage.setItem(this.storageKey, JSON.stringify(storageData))
      
      return {
        imported: newVaults.length,
        skipped: validatedVaults.length - newVaults.length,
        total: mergedVaults.length
      }
    } catch (error) {
      console.error('Failed to import vaults:', error)
      throw new Error('Failed to import vault data')
    }
  }

  clearAllVaults() {
    localStorage.removeItem(this.storageKey)
  }

  getStorageStats() {
    const data = this.getStorageData()
    const vaults = data.vaults || []
    
    return {
      totalVaults: vaults.length,
      totalBalance: vaults.reduce((sum, vault) => sum + (vault.balance || 0), 0),
      lastUpdated: data.lastUpdated,
      version: data.version,
      storageSize: new Blob([JSON.stringify(data)]).size
    }
  }
}
