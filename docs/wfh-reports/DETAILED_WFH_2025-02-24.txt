// Enhanced auto-withdrawal service with monitoring
export class AutoWithdrawalService {
  constructor() {
    this.monitoringInterval = null
    this.priceCheckInterval = 30000 // 30 seconds
    this.activeVaults = new Map() // vaultId -> vault data
    this.priceHistory = []
    this.isMonitoring = false
    this.serviceStatus = 'stopped'
    this.lastPriceCheck = null
    this.withdrawalQueue = []
    this.processingWithdrawals = false
  }

  init(store) {
    this.store = store
    this.loadConfiguration()
    console.log('AutoWithdrawalService initialized')
  }

  loadConfiguration() {
    // Load configuration from store or environment
    const config = this.store?.state?.autoWithdrawal || {
      enabled: true,
      priceCheckInterval: 30000,
      maxRetries: 3,
      withdrawalDelay: 5000
    }
    
    this.priceCheckInterval = config.priceCheckInterval || 30000
    this.maxRetries = config.maxRetries || 3
    this.withdrawalDelay = config.withdrawalDelay || 5000
  }

  startMonitoring() {
    if (this.isMonitoring) {
      console.log('Auto-withdrawal monitoring already running')
      return
    }

    console.log('Starting auto-withdrawal monitoring...')
    this.isMonitoring = true
    this.serviceStatus = 'running'
    
    // Load active vaults
    this.loadActiveVaults()
    
    // Start price monitoring
    this.startPriceMonitoring()
    
    // Process withdrawal queue
    this.processWithdrawalQueue()
    
    // Emit service status
    this.emitServiceStatus('started')
  }

  stopMonitoring() {
    if (!this.isMonitoring) {
      console.log('Auto-withdrawal monitoring not running')
      return
    }

    console.log('Stopping auto-withdrawal monitoring...')
    this.isMonitoring = false
    this.serviceStatus = 'stopped'
    
    // Clear intervals
    if (this.monitoringInterval) {
      clearInterval(this.monitoringInterval)
      this.monitoringInterval = null
    }
    
    // Clear active vaults
    this.activeVaults.clear()
    
    // Emit service status
    this.emitServiceStatus('stopped')
  }

  loadActiveVaults() {
    try {
      // Load vaults from storage
      const vaultStorage = require('./vault-storage')
      const vaults = vaultStorage.getAllVaults()
      
      // Filter vaults with auto-withdrawal enabled
      const autoWithdrawalVaults = vaults.filter(vault => 
        vault.autoWithdrawal && vault.status === 'active'
      )
      
      // Add to active vaults map
      autoWithdrawalVaults.forEach(vault => {
        this.activeVaults.set(vault.id, {
          ...vault,
          lastChecked: null,
          lastWithdrawalAttempt: null,
          retryCount: 0
        })
      })
      
      console.log(`Loaded ${autoWithdrawalVaults.length} vaults for auto-withdrawal`)
    } catch (error) {
      console.error('Failed to load active vaults:', error)
    }
  }

  startPriceMonitoring() {
    this.monitoringInterval = setInterval(async () => {
      try {
        await this.checkPriceAndProcessWithdrawals()
      } catch (error) {
        console.error('Price monitoring error:', error)
        this.handleMonitoringError(error)
      }
    }, this.priceCheckInterval)
    
    console.log(`Price monitoring started with ${this.priceCheckInterval}ms interval`)
  }

  async checkPriceAndProcessWithdrawals() {
    try {
      // Get current price
      const priceData = await this.getCurrentPrice()
      
      if (!priceData || !priceData.price) {
        console.warn('Failed to get current price, skipping this check')
        return
      }
      
      const currentPrice = priceData.price
      this.lastPriceCheck = Date.now()
      
      // Add to price history
      this.addToPriceHistory(currentPrice)
      
      // Check all active vaults
      await this.checkAllVaults(currentPrice)
      
      // Emit price update
      this.emitPriceUpdate(currentPrice)
      
    } catch (error) {
      console.error('Price check failed:', error)
      this.handleMonitoringError(error)
    }
  }

  async getCurrentPrice() {
    try {
      const { fetchOraclePrice } = require('./oracle')
      return await fetchOraclePrice()
    } catch (error) {
      console.error('Failed to fetch current price:', error)
      throw error
    }
  }

  addToPriceHistory(price) {
    const priceEntry = {
      price: price,
      timestamp: Date.now(),
      source: 'oracle'
    }
    
    this.priceHistory.push(priceEntry)
    
    // Keep only last 100 entries
    if (this.priceHistory.length > 100) {
      this.priceHistory = this.priceHistory.slice(-100)
    }
  }

  async checkAllVaults(currentPrice) {
    const vaultsToProcess = []
    
    // Check each active vault
    for (const [vaultId, vaultData] of this.activeVaults) {
      try {
        const shouldWithdraw = await this.checkVaultCondition(vaultData, currentPrice)
        
        if (shouldWithdraw) {
          vaultsToProcess.push({
            vaultId: vaultId,
            vaultData: vaultData,
            currentPrice: currentPrice,
            reason: 'price_target_reached'
          })
        }
      } catch (error) {
        console.error(`Error checking vault ${vaultId}:`, error)
      }
    }
    
    // Add vaults to withdrawal queue
    if (vaultsToProcess.length > 0) {
      console.log(`Found ${vaultsToProcess.length} vaults ready for withdrawal`)
      vaultsToProcess.forEach(vault => {
        this.addToWithdrawalQueue(vault)
      })
    }
  }

  async checkVaultCondition(vaultData, currentPrice) {
    // Check if price target is reached
    if (currentPrice >= vaultData.priceTarget) {
      // Check if already withdrawn recently (prevent duplicate withdrawals)
      const lastWithdrawal = vaultData.lastWithdrawalAttempt
      const cooldownPeriod = 60000 // 1 minute cooldown
      
      if (lastWithdrawal && (Date.now() - lastWithdrawal < cooldownPeriod)) {
        console.log(`Vault ${vaultData.id} in cooldown period, skipping`)
        return false
      }
      
      // Check retry count
      if (vaultData.retryCount >= this.maxRetries) {
        console.log(`Vault ${vaultData.id} exceeded max retries, skipping`)
        return false
      }
      
      return true
    }
    
    return false
  }

  addToWithdrawalQueue(vault) {
    // Check if already in queue
    const existingIndex = this.withdrawalQueue.findIndex(item => item.vaultId === vault.vaultId)
    
    if (existingIndex === -1) {
      this.withdrawalQueue.push(vault)
      console.log(`Added vault ${vault.vaultId} to withdrawal queue`)
    }
  }

  async processWithdrawalQueue() {
    if (this.processingWithdrawals) {
      return
    }
    
    this.processingWithdrawals = true
    
    while (this.withdrawalQueue.length > 0 && this.isMonitoring) {
      const vault = this.withdrawalQueue.shift()
      
      try {
        await this.processWithdrawal(vault)
      } catch (error) {
        console.error(`Failed to process withdrawal for vault ${vault.vaultId}:`, error)
        this.handleWithdrawalError(vault, error)
      }
      
      // Small delay between withdrawals
      if (this.withdrawalQueue.length > 0) {
        await this.delay(this.withdrawalDelay)
      }
    }
    
    this.processingWithdrawals = false
  }

  async processWithdrawal(vault) {
    console.log(`Processing withdrawal for vault ${vault.vaultId} at price $${vault.currentPrice}`)
    
    try {
      // Update vault status
      const vaultData = this.activeVaults.get(vault.vaultId)
      vaultData.lastWithdrawalAttempt = Date.now()
      
      // Execute withdrawal
      const result = await this.executeWithdrawal(vault)
      
      if (result.success) {
        console.log(`Withdrawal successful for vault ${vault.vaultId}`)
        
        // Update vault status
        vaultData.status = 'completed'
        vaultData.completedAt = Date.now()
        vaultData.withdrawalPrice = vault.currentPrice
        vaultData.withdrawalTxId = result.txId
        
        // Remove from active vaults
        this.activeVaults.delete(vault.vaultId)
        
        // Emit withdrawal success
        this.emitWithdrawalSuccess(vault, result)
        
        // Update vault storage
        await this.updateVaultInStorage(vault.vaultId, {
          status: 'completed',
          completedAt: Date.now(),
          withdrawalPrice: vault.currentPrice,
          withdrawalTxId: result.txId
        })
        
      } else {
        throw new Error(result.error || 'Withdrawal failed')
      }
      
    } catch (error) {
      console.error(`Withdrawal processing failed for vault ${vault.vaultId}:`, error)
      
      // Update retry count
      const vaultData = this.activeVaults.get(vault.vaultId)
      vaultData.retryCount += 1
      
      // Re-queue if retries remaining
      if (vaultData.retryCount < this.maxRetries) {
        console.log(`Re-queuing vault ${vault.vaultId} for retry (${vaultData.retryCount}/${this.maxRetries})`)
        
        // Add back to queue with delay
        setTimeout(() => {
          this.addToWithdrawalQueue(vault)
        }, this.withdrawalDelay * vaultData.retryCount)
      } else {
        console.log(`Vault ${vault.vaultId} exceeded max retries, marking as failed`)
        
        // Mark as failed
        vaultData.status = 'failed'
        this.activeVaults.delete(vault.vaultId)
        
        // Update vault storage
        await this.updateVaultInStorage(vault.vaultId, {
          status: 'failed',
          failedAt: Date.now(),
          failureReason: error.message
        })
        
        // Emit withdrawal failure
        this.emitWithdrawalFailure(vault, error)
      }
      
      throw error
    }
  }

  async executeWithdrawal(vault) {
    try {
      // Import blockchain service
      const blockchainService = require('./blockchain')
      
      // Execute withdrawal transaction
      const result = await blockchainService.withdrawFromVault(
        vault.vaultData.contractAddress,
        vault.vaultData.ownerPkhHex,
        vault.vaultData.oraclePkHex,
        vault.vaultData.priceTargetCents,
        vault.vaultData.vaultSalt
      )
      
      return {
        success: true,
        txId: result.txId,
        transaction: result.transaction
      }
      
    } catch (error) {
      console.error('Withdrawal execution failed:', error)
      return {
        success: false,
        error: error.message
      }
    }
  }

  async updateVaultInStorage(vaultId, updates) {
    try {
      const vaultStorage = require('./vault-storage')
      await vaultStorage.updateVault(vaultId, updates)
    } catch (error) {
      console.error('Failed to update vault in storage:', error)
    }
  }

  handleMonitoringError(error) {
    console.error('Monitoring error:', error)
    
    // Emit error event
    this.emitServiceError(error)
    
    // Continue monitoring unless critical error
    if (error.critical) {
      console.error('Critical monitoring error, stopping service')
      this.stopMonitoring()
    }
  }

  handleWithdrawalError(vault, error) {
    console.error(`Withdrawal error for vault ${vault.vaultId}:`, error)
    
    // Emit withdrawal error
    this.emitWithdrawalError(vault, error)
  }

  // Event emitters
  emitServiceStatus(status) {
    const event = new CustomEvent('auto-withdrawal:status', {
      detail: { status, timestamp: Date.now() }
    })
    window.dispatchEvent(event)
  }

  emitPriceUpdate(price) {
    const event = new CustomEvent('auto-withdrawal:price', {
      detail: { price, timestamp: Date.now() }
    })
    window.dispatchEvent(event)
  }

  emitWithdrawalSuccess(vault, result) {
    const event = new CustomEvent('auto-withdrawal:success', {
      detail: { vault, result, timestamp: Date.now() }
    })
    window.dispatchEvent(event)
  }

  emitWithdrawalFailure(vault, error) {
    const event = new CustomEvent('auto-withdrawal:failure', {
      detail: { vault, error, timestamp: Date.now() }
    })
    window.dispatchEvent(event)
  }

  emitWithdrawalError(vault, error) {
    const event = new CustomEvent('auto-withdrawal:error', {
      detail: { vault, error, timestamp: Date.now() }
    })
    window.dispatchEvent(event)
  }

  emitServiceError(error) {
    const event = new CustomEvent('auto-withdrawal:service-error', {
      detail: { error, timestamp: Date.now() }
    })
    window.dispatchEvent(event)
  }

  // Utility methods
  delay(ms) {
    return new Promise(resolve => setTimeout(resolve, ms))
  }

  // Public API methods
  getServiceStatus() {
    return {
      isMonitoring: this.isMonitoring,
      status: this.serviceStatus,
      activeVaults: this.activeVaults.size,
      queueLength: this.withdrawalQueue.length,
      lastPriceCheck: this.lastPriceCheck,
      priceCheckInterval: this.priceCheckInterval
    }
  }

  getPriceHistory(limit = 10) {
    return this.priceHistory.slice(-limit)
  }

  getActiveVaults() {
    return Array.from(this.activeVaults.values())
  }

  addVault(vault) {
    if (vault.autoWithdrawal && vault.status === 'active') {
      this.activeVaults.set(vault.id, {
        ...vault,
        lastChecked: null,
        lastWithdrawalAttempt: null,
        retryCount: 0
      })
      
      console.log(`Added vault ${vault.id} to auto-withdrawal monitoring`)
    }
  }

  removeVault(vaultId) {
    const removed = this.activeVaults.delete(vaultId)
    
    if (removed) {
      console.log(`Removed vault ${vaultId} from auto-withdrawal monitoring`)
    }
    
    // Remove from queue if present
    this.withdrawalQueue = this.withdrawalQueue.filter(item => item.vaultId !== vaultId)
    
    return removed
  }

  updateConfiguration(config) {
    this.priceCheckInterval = config.priceCheckInterval || this.priceCheckInterval
    this.maxRetries = config.maxRetries || this.maxRetries
    this.withdrawalDelay = config.withdrawalDelay || this.withdrawalDelay
    
    // Restart monitoring if interval changed and service is running
    if (config.priceCheckInterval && this.isMonitoring) {
      this.stopMonitoring()
      this.startMonitoring()
    }
    
    console.log('Auto-withdrawal configuration updated')
  }
}
