// Enhanced security service with comprehensive validation
export class SecurityService {
  constructor() {
    this.rateLimits = new Map() // endpoint -> limit data
    this.deviceFingerprint = null
    this.sessionData = new Map()
    this.securityConfig = {
      maxRequestsPerMinute: 60,
      maxTransactionsPerHour: 10,
      sessionTimeout: 3600000, // 1 hour
      maxFailedAttempts: 5,
      lockoutDuration: 900000 // 15 minutes
    }
    this.lockedAccounts = new Map()
  }

  async initialize() {
    // Generate device fingerprint
    this.deviceFingerprint = await this.generateDeviceFingerprint()
    
    // Initialize rate limiting
    this.initializeRateLimiting()
    
    console.log('Security service initialized')
  }

  async generateDeviceFingerprint() {
    try {
      const canvas = document.createElement('canvas')
      const ctx = canvas.getContext('2d')
      ctx.textBaseline = 'top'
      ctx.font = '14px Arial'
      ctx.fillText('Device fingerprint', 2, 2)
      
      const canvasFingerprint = canvas.toDataURL()
      
      const fingerprint = {
        userAgent: navigator.userAgent,
        language: navigator.language,
        platform: navigator.platform,
        screenResolution: `${screen.width}x${screen.height}`,
        colorDepth: screen.colorDepth,
        timezone: Intl.DateTimeFormat().resolvedOptions().timeZone,
        canvas: canvasFingerprint,
        webgl: this.getWebGLFingerprint(),
        fonts: this.getFontsFingerprint()
      }
      
      // Create hash from fingerprint data
      const fingerprintString = JSON.stringify(fingerprint)
      const hash = await this.hashString(fingerprintString)
      
      return hash
    } catch (error) {
      console.error('Failed to generate device fingerprint:', error)
      return 'fallback_fingerprint_' + Date.now()
    }
  }

  getWebGLFingerprint() {
    try {
      const canvas = document.createElement('canvas')
      const gl = canvas.getContext('webgl') || canvas.getContext('experimental-webgl')
      
      if (!gl) return null
      
      const debugInfo = gl.getExtension('WEBGL_debug_renderer_info')
      const renderer = debugInfo ? gl.getParameter(debugInfo.UNMASKED_RENDERER_WEBGL) : null
      const vendor = debugInfo ? gl.getParameter(debugInfo.UNMASKED_VENDOR_WEBGL) : null
      
      return { renderer, vendor }
    } catch (error) {
      return null
    }
  }

  getFontsFingerprint() {
    try {
      const fonts = [
        'Arial', 'Verdana', 'Times New Roman', 'Courier New', 'Georgia',
        'Palatino', 'Garamond', 'Comic Sans MS', 'Trebuchet MS', 'Arial Black'
      ]
      
      const availableFonts = []
      const testString = 'mmmmmmmmmmlli'
      
      const canvas = document.createElement('canvas')
      const ctx = canvas.getContext('2d')
      ctx.textBaseline = 'top'
      ctx.font = '72px monospace'
      ctx.fillText(testString, 0, 0)
      
      const baseline = ctx.getImageData(0, 0, canvas.width, canvas.height).data
      
      fonts.forEach(font => {
        ctx.font = `72px ${font}, monospace`
        ctx.fillText(testString, 0, 0)
        const fontData = ctx.getImageData(0, 0, canvas.width, canvas.height).data
        
        const isDifferent = baseline.some((value, index) => value !== fontData[index])
        if (isDifferent) {
          availableFonts.push(font)
        }
      })
      
      return availableFonts
    } catch (error) {
      return []
    }
  }

  async hashString(str) {
    try {
      const encoder = new TextEncoder()
      const data = encoder.encode(str)
      const hashBuffer = await crypto.subtle.digest('SHA-256', data)
      const hashArray = Array.from(new Uint8Array(hashBuffer))
      return hashArray.map(b => b.toString(16).padStart(2, '0')).join('')
    } catch (error) {
      // Fallback hash function
      let hash = 0
      for (let i = 0; i < str.length; i++) {
        const char = str.charCodeAt(i)
        hash = ((hash << 5) - hash) + char
        hash = hash & hash // Convert to 32-bit integer
      }
      return hash.toString(16)
    }
  }

  initializeRateLimiting() {
    // Initialize rate limits for common endpoints
    this.rateLimits.set('api/oracle', {
      requests: [],
      maxRequests: this.securityConfig.maxRequestsPerMinute,
      windowMs: 60000 // 1 minute
    })
    
    this.rateLimits.set('api/transaction', {
      requests: [],
      maxRequests: this.securityConfig.maxTransactionsPerHour,
      windowMs: 3600000 // 1 hour
    })
    
    this.rateLimits.set('api/auth', {
      requests: [],
      maxRequests: 5, // Lower limit for auth
      windowMs: 60000 // 1 minute
    })
  }

  async checkRateLimit(endpoint, identifier = this.deviceFingerprint) {
    const limit = this.rateLimits.get(endpoint)
    
    if (!limit) {
      return { allowed: true }
    }
    
    const now = Date.now()
    const windowStart = now - limit.windowMs
    
    // Clean old requests
    limit.requests = limit.requests.filter(timestamp => timestamp > windowStart)
    
    // Check if limit exceeded
    if (limit.requests.length >= limit.maxRequests) {
      return {
        allowed: false,
        remaining: 0,
        resetTime: limit.requests[0] + limit.windowMs,
        retryAfter: Math.ceil((limit.requests[0] + limit.windowMs - now) / 1000)
      }
    }
    
    // Add current request
    limit.requests.push(now)
    
    return {
      allowed: true,
      remaining: limit.maxRequests - limit.requests.length,
      resetTime: now + limit.windowMs
    }
  }

  validateInput(input, type, options = {}) {
    const validationRules = {
      price: {
        pattern: /^\d+(\.\d{1,8})?$/,
        min: 0.01,
        max: 10000,
        maxLength: 20
      },
      address: {
        pattern: /^(bitcoincash|bchtest|chipnet):[a-zA-Z0-9]+$/,
        minLength: 20,
        maxLength: 100
      },
      amount: {
        pattern: /^\d+(\.\d{1,8})?$/,
        min: 0.00000001,
        max: 1000000,
        maxLength: 20
      },
      text: {
        maxLength: options.maxLength || 1000,
        allowEmpty: options.allowEmpty || false
      },
      email: {
        pattern: /^[^\s@]+@[^\s@]+\.[^\s@]+$/,
        maxLength: 254
      }
    }
    
    const rules = validationRules[type]
    if (!rules) {
      return { valid: false, error: 'Unknown validation type' }
    }
    
    // Check if input exists
    if (!input && !rules.allowEmpty) {
      return { valid: false, error: 'Input is required' }
    }
    
    if (!input && rules.allowEmpty) {
      return { valid: true }
    }
    
    const inputStr = String(input)
    
    // Check length
    if (rules.minLength && inputStr.length < rules.minLength) {
      return { valid: false, error: `Input must be at least ${rules.minLength} characters` }
    }
    
    if (rules.maxLength && inputStr.length > rules.maxLength) {
      return { valid: false, error: `Input must be no more than ${rules.maxLength} characters` }
    }
    
    // Check pattern
    if (rules.pattern && !rules.pattern.test(inputStr)) {
      return { valid: false, error: 'Input format is invalid' }
    }
    
    // Check numeric ranges
    if (rules.min || rules.max) {
      const numValue = parseFloat(inputStr)
      
      if (isNaN(numValue)) {
        return { valid: false, error: 'Input must be a number' }
      }
      
      if (rules.min !== undefined && numValue < rules.min) {
        return { valid: false, error: `Input must be at least ${rules.min}` }
      }
      
      if (rules.max !== undefined && numValue > rules.max) {
        return { valid: false, error: `Input must be no more than ${rules.max}` }
      }
    }
    
    // Sanitize input
    const sanitized = this.sanitizeInput(inputStr, type)
    
    return { valid: true, sanitized: sanitized }
  }

  sanitizeInput(input, type) {
    // Basic sanitization
    let sanitized = input.trim()
    
    // Remove potentially dangerous characters
    sanitized = sanitized.replace(/[<>]/g, '')
    
    // Type-specific sanitization
    switch (type) {
      case 'text':
        // Allow only safe characters
        sanitized = sanitized.replace(/[^\w\s\-.,!?@#$%^&*()]/g, '')
        break
      case 'price':
      case 'amount':
        // Ensure only numeric characters and decimal point
        sanitized = sanitized.replace(/[^\d.]/g, '')
        // Remove multiple decimal points
        sanitized = sanitized.replace(/\.+/g, '.')
        break
      case 'address':
        // Already validated with regex, just normalize case
        sanitized = sanitized.toLowerCase()
        break
    }
    
    return sanitized
  }

  async validateTransaction(user, transaction) {
    try {
      // Check rate limit
      const rateLimitCheck = await this.checkRateLimit('api/transaction', user.id || this.deviceFingerprint)
      
      if (!rateLimitCheck.allowed) {
        throw new Error(`Rate limit exceeded. Try again in ${rateLimitCheck.retryAfter} seconds`)
      }
      
      // Validate transaction structure
      const validationResult = this.validateTransactionStructure(transaction)
      
      if (!validationResult.valid) {
        throw new Error(`Invalid transaction: ${validationResult.error}`)
      }
      
      // Check for replay attacks
      const replayCheck = await this.checkReplayAttack(transaction)
      
      if (replayCheck.isReplay) {
        throw new Error('Potential replay attack detected')
      }
      
      // Verify transaction signature
      const signatureCheck = await this.verifyTransactionSignature(transaction, user)
      
      if (!signatureCheck.valid) {
        throw new Error('Invalid transaction signature')
      }
      
      // Update session data
      this.updateSessionData(user.id, {
        lastTransaction: Date.now(),
        transactionCount: (this.getSessionData(user.id).transactionCount || 0) + 1
      })
      
      return {
        valid: true,
        warnings: validationResult.warnings || []
      }
      
    } catch (error) {
      console.error('Transaction validation failed:', error)
      
      // Log security event
      this.logSecurityEvent('transaction_validation_failed', {
        user: user.id,
        error: error.message,
        transaction: this.hashTransaction(transaction)
      })
      
      throw error
    }
  }

  validateTransactionStructure(transaction) {
    const requiredFields = ['raw', 'signature', 'inputs', 'outputs']
    
    for (const field of requiredFields) {
      if (!transaction[field]) {
        return { valid: false, error: `Missing required field: ${field}` }
      }
    }
    
    // Validate transaction format
    if (typeof transaction.raw !== 'string' || !transaction.raw.startsWith('02')) {
      return { valid: false, error: 'Invalid transaction format' }
    }
    
    // Validate inputs
    if (!Array.isArray(transaction.inputs) || transaction.inputs.length === 0) {
      return { valid: false, error: 'Invalid transaction inputs' }
    }
    
    // Validate outputs
    if (!Array.isArray(transaction.outputs) || transaction.outputs.length === 0) {
      return { valid: false, error: 'Invalid transaction outputs' }
    }
    
    // Check output amounts
    const totalOutput = transaction.outputs.reduce((sum, output) => {
      if (typeof output.value !== 'number' || output.value < 0) {
        throw new Error('Invalid output value')
      }
      return sum + output.value
    }, 0)
    
    if (totalOutput <= 0) {
      return { valid: false, error: 'Transaction must have positive output value' }
    }
    
    return { valid: true }
  }

  async checkReplayAttack(transaction) {
    try {
      // Create transaction hash
      const transactionHash = await this.hashTransaction(transaction)
      
      // Check if transaction hash exists in recent transactions
      const recentTransactions = this.getRecentTransactions()
      const isReplay = recentTransactions.includes(transactionHash)
      
      if (isReplay) {
        this.logSecurityEvent('replay_attack_detected', {
          transactionHash,
          timestamp: Date.now()
        })
      }
      
      // Add to recent transactions
      this.addRecentTransaction(transactionHash)
      
      return { isReplay }
    } catch (error) {
      console.error('Replay attack check failed:', error)
      return { isReplay: false, error: error.message }
    }
  }

  async hashTransaction(transaction) {
    const transactionString = JSON.stringify({
      raw: transaction.raw,
      inputs: transaction.inputs,
      outputs: transaction.outputs
    })
    
    return await this.hashString(transactionString)
  }

  getRecentTransactions() {
    const key = 'recent_transactions'
    const data = localStorage.getItem(key)
    
    if (!data) {
      return []
    }
    
    try {
      const parsed = JSON.parse(data)
      const now = Date.now()
      
      // Filter to last 24 hours
      return parsed.filter(item => now - item.timestamp < 86400000).map(item => item.hash)
    } catch (error) {
      console.error('Failed to parse recent transactions:', error)
      return []
    }
  }

  addRecentTransaction(hash) {
    const key = 'recent_transactions'
    const data = localStorage.getItem(key)
    
    let transactions = []
    
    if (data) {
      try {
        transactions = JSON.parse(data)
      } catch (error) {
        console.error('Failed to parse recent transactions:', error)
      }
    }
    
    // Add new transaction
    transactions.push({
      hash: hash,
      timestamp: Date.now()
    })
    
    // Keep only last 1000 transactions
    if (transactions.length > 1000) {
      transactions = transactions.slice(-1000)
    }
    
    localStorage.setItem(key, JSON.stringify(transactions))
  }

  async verifyTransactionSignature(transaction, user) {
    try {
      // This would integrate with the actual BCH signature verification
      // For now, implement basic validation
      
      if (!transaction.signature || typeof transaction.signature !== 'string') {
        return { valid: false, error: 'Invalid signature format' }
      }
      
      if (transaction.signature.length < 64) {
        return { valid: false, error: 'Signature too short' }
      }
      
      // Verify signature matches transaction (simplified)
      const expectedSigner = user.address || user.publicKey
      
      if (!expectedSigner) {
        return { valid: false, error: 'No signer information available' }
      }
      
      // In a real implementation, this would use BCH cryptography libraries
      // For now, just check basic format
      return { valid: true }
      
    } catch (error) {
      console.error('Signature verification failed:', error)
      return { valid: false, error: error.message }
    }
  }

  updateSessionData(userId, data) {
    const sessionData = this.getSessionData(userId)
    Object.assign(sessionData, data)
    sessionData.lastUpdated = Date.now()
    this.sessionData.set(userId, sessionData)
  }

  getSessionData(userId) {
    if (!this.sessionData.has(userId)) {
      this.sessionData.set(userId, {
        userId: userId,
        createdAt: Date.now(),
        lastUpdated: Date.now(),
        transactionCount: 0,
        failedAttempts: 0
      })
    }
    return this.sessionData.get(userId)
  }

  checkAccountLock(userId) {
    const lockData = this.lockedAccounts.get(userId)
    
    if (!lockData) {
      return { locked: false }
    }
    
    const now = Date.now()
    
    if (now - lockData.lockedAt > this.securityConfig.lockoutDuration) {
      // Lockout expired
      this.lockedAccounts.delete(userId)
      return { locked: false }
    }
    
    return {
      locked: true,
      remainingTime: Math.ceil((lockData.lockedAt + this.securityConfig.lockoutDuration - now) / 1000),
      reason: lockData.reason
    }
  }

  lockAccount(userId, reason = 'Too many failed attempts') {
    this.lockedAccounts.set(userId, {
      lockedAt: Date.now(),
      reason: reason
    })
    
    this.logSecurityEvent('account_locked', {
      userId: userId,
      reason: reason,
      timestamp: Date.now()
    })
  }

  logSecurityEvent(eventType, data) {
    const event = {
      type: eventType,
      data: data,
      timestamp: Date.now(),
      deviceFingerprint: this.deviceFingerprint,
      userAgent: navigator.userAgent
    }
    
    console.warn('Security event:', event)
    
    // Store security events
    const key = 'security_events'
    const events = JSON.parse(localStorage.getItem(key) || '[]')
    events.push(event)
    
    // Keep only last 100 events
    if (events.length > 100) {
      events.splice(0, events.length - 100)
    }
    
    localStorage.setItem(key, JSON.stringify(events))
  }

  // Public API methods
  getSecurityStatus() {
    return {
      deviceFingerprint: this.deviceFingerprint,
      activeSessions: this.sessionData.size,
      lockedAccounts: this.lockedAccounts.size,
      rateLimits: Array.from(this.rateLimits.entries()).map(([endpoint, limit]) => ({
        endpoint,
        currentRequests: limit.requests.length,
        maxRequests: limit.maxRequests
      }))
    }
  }

  clearSessionData(userId) {
    this.sessionData.delete(userId)
  }

  clearAllSessionData() {
    this.sessionData.clear()
    this.lockedAccounts.clear()
  }
}
