// Enhanced signing with retry logic
const signTransactionWithRetry = async (rawTransaction, maxRetries = 3) => {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      const wc = this.$walletConnect
      if (!wc || !wc.isConnected()) {
        throw new Error('Wallet not connected')
      }

      // Validate transaction before signing
      await this.validateTransactionFormat(rawTransaction)
      
      const result = await wc.request('bch_signTransaction', {
        transaction: rawTransaction
      })

      // Verify signature result
      if (!result || !result.signature) {
        throw new Error('Invalid signature result from wallet')
      }

      return result
    } catch (error) {
      console.error(`Transaction signing attempt ${attempt} failed:`, error)
      
      if (attempt === maxRetries) {
        // Last attempt failed, try alternative signing method
        return await this.tryAlternativeSigning(rawTransaction)
      }
      
      // Exponential backoff before retry
      const delay = Math.pow(2, attempt) * 1000
      await new Promise(resolve => setTimeout(resolve, delay))
    }
  }
}

// Transaction validation before signing
const validateTransactionFormat = async (rawTransaction) => {
  if (!rawTransaction || typeof rawTransaction !== 'string') {
    throw new Error('Invalid transaction format')
  }
  
  // Validate transaction structure
  try {
    const tx = bitbox.Transaction.fromHex(rawTransaction)
    if (!tx || !tx.outputs || tx.outputs.length === 0) {
      throw new Error('Invalid transaction structure')
    }
  } catch (error) {
    throw new Error(`Transaction validation failed: ${error.message}`)
  }
}

// Alternative signing method as fallback
const tryAlternativeSigning = async (rawTransaction) => {
  try {
    const wc = this.$walletConnect
    
    // Try alternative signing method
    const result = await wc.request('bch_signTransaction', {
      data: rawTransaction,
      type: 'transaction'
    })
    
    return result
  } catch (error) {
    console.error('Alternative signing method failed:', error)
    throw new Error('All signing methods failed')
  }
}
