// Enhanced transaction monitoring with adaptive polling
class TransactionMonitor {
  constructor() {
    this.monitoringIntervals = new Map() // txId -> interval
    this.statusCache = new Map() // txId -> status
    this.basePollingInterval = 5000 // 5 seconds base
  }

  async startMonitoring(txId, callback) {
    // Clear existing monitoring if any
    this.stopMonitoring(txId)
    
    let pollingInterval = this.basePollingInterval
    let consecutiveErrors = 0
    
    const monitor = async () => {
      try {
        // Check cache first
        if (this.statusCache.has(txId)) {
          const cachedStatus = this.statusCache.get(txId)
          const cacheAge = Date.now() - cachedStatus.timestamp
          
          // Use cached status if less than 30 seconds old
          if (cacheAge < 30000 && cachedStatus.status !== 'pending') {
            callback(cachedStatus)
            return
          }
        }

        // Fetch current status
        const status = await this.fetchTransactionStatus(txId)
        
        // Update cache
        this.statusCache.set(txId, {
          ...status,
          timestamp: Date.now()
        })
        
        // Adaptive polling based on status
        if (status.status === 'confirmed') {
          this.stopMonitoring(txId)
          callback(status)
          return
        } else if (status.status === 'failed') {
          pollingInterval = Math.min(pollingInterval * 2, 30000) // Slower polling for failed tx
        } else if (status.status === 'pending') {
          const age = Date.now() - status.timestamp
          // Faster polling for new transactions, slower for older ones
          if (age < 60000) { // Less than 1 minute old
            pollingInterval = this.basePollingInterval
          } else if (age < 300000) { // Less than 5 minutes old
            pollingInterval = this.basePollingInterval * 2
          } else {
            pollingInterval = Math.min(pollingInterval * 1.5, 15000)
          }
        }
        
        consecutiveErrors = 0 // Reset error counter on success
        callback(status)
        
      } catch (error) {
        console.error(`Transaction monitoring error for ${txId}:`, error)
        consecutiveErrors++
        
        // Exponential backoff on errors
        if (consecutiveErrors > 3) {
          pollingInterval = Math.min(pollingInterval * 2, 60000)
        }
        
        // Continue monitoring despite errors
        if (consecutiveErrors >= 10) {
          console.warn(`Stopping monitoring for ${txId} after too many errors`)
          this.stopMonitoring(txId)
          return
        }
      }
      
      // Schedule next check with adaptive interval
      this.monitoringIntervals.set(txId, setTimeout(monitor, pollingInterval))
    }
    
    // Start monitoring
    monitor()
  }

  stopMonitoring(txId) {
    const interval = this.monitoringIntervals.get(txId)
    if (interval) {
      clearTimeout(interval)
      this.monitoringIntervals.delete(txId)
    }
  }

  async fetchTransactionStatus(txId) {
    try {
      // Try multiple blockchain explorers for redundancy
      const explorers = [
        'https://chipnet.bch.ninja/api/tx/',
        'https://chipnet.fullstack.cash/api/tx/'
      ]
      
      for (const explorer of explorers) {
        try {
          const response = await fetch(`${explorer}${txId}`)
          if (response.ok) {
            const txData = await response.json()
            return {
              txId: txId,
              status: txData.blockheight ? 'confirmed' : 'pending',
              confirmations: txData.confirmations || 0,
              blockheight: txData.blockheight || null,
              timestamp: txData.time ? txData.time * 1000 : Date.now()
            }
          }
        } catch (explorerError) {
          console.warn(`Explorer ${explorer} failed:`, explorerError)
          continue
        }
      }
      
      throw new Error('All explorers failed')
    } catch (error) {
      return {
        txId: txId,
        status: 'unknown',
        error: error.message,
        timestamp: Date.now()
      }
    }
  }

  clearCache() {
    this.statusCache.clear()
  }
}

const transactionMonitor = new TransactionMonitor()

// Start monitoring a transaction
transactionMonitor.startMonitoring(txId, (status) => {
  console.log(`Transaction ${txId} status:`, status)
  
  if (status.status === 'confirmed') {
    // Handle confirmed transaction
    this.handleConfirmedTransaction(status)
  } else if (status.status === 'failed') {
    // Handle failed transaction
    this.handleFailedTransaction(status)
  }
})
