// Enhanced oracle service with multiple fallbacks
export class OracleService {
  constructor() {
    this.priceSources = [
      { name: 'oracles.cash', priority: 1, url: 'https://oracles.generalprotocols.com/api/v1/oracles' },
      { name: 'coingecko', priority: 2, url: 'https://api.coingecko.com/api/v3/simple/price' },
      { name: 'coinmarketcap', priority: 3, url: 'https://pro-api.coinmarketcap.com/v1/cryptocurrency/quotes/latest' }
    ]
    this.sourceHealth = new Map()
    this.priceCache = new Map()
    this.cacheTimeout = 30000 // 30 seconds
  }

  async fetchPriceWithFallbacks() {
    const cacheKey = 'current_price'
    
    // Check cache first
    if (this.priceCache.has(cacheKey)) {
      const cached = this.priceCache.get(cacheKey)
      if (Date.now() - cached.timestamp < this.cacheTimeout) {
        console.log('Using cached price:', cached.price)
        return cached.data
      }
    }

    let lastError = null
    const priceData = []
    
    // Try all sources in priority order
    for (const source of this.priceSources) {
      try {
        console.log(`Attempting to fetch price from ${source.name}`)
        const price = await this.fetchFromSource(source)
        
        if (price && this.validatePrice(price)) {
          priceData.push({
            source: source.name,
            price: price,
            timestamp: Date.now()
          })
          
          // Update source health
          this.updateSourceHealth(source.name, 'healthy')
        }
      } catch (error) {
        console.warn(`Failed to fetch from ${source.name}:`, error.message)
        lastError = error
        this.updateSourceHealth(source.name, 'unhealthy')
      }
    }

    if (priceData.length === 0) {
      // All sources failed, use fallback
      console.error('All price sources failed, using fallback')
      return this.getFallbackPrice(lastError)
    }

    // Analyze price data for consistency
    const consensusPrice = this.calculateConsensusPrice(priceData)
    
    if (!consensusPrice) {
      console.warn('Price consensus not reached, using most reliable source')
      return priceData[0] // Use first successful source
    }

    // Cache the result
    const result = {
      price: consensusPrice.price,
      source: consensusPrice.source,
      sources: priceData,
      timestamp: Date.now(),
      status: 'success'
    }
    
    this.priceCache.set(cacheKey, {
      data: result,
      timestamp: Date.now()
    })
    
    return result
  }

  async fetchFromSource(source) {
    switch (source.name) {
      case 'oracles.cash':
        return await this.fetchFromOraclesCash(source)
      case 'coingecko':
        return await this.fetchFromCoinGecko(source)
      case 'coinmarketcap':
        return await this.fetchFromCoinMarketCap(source)
      default:
        throw new Error(`Unknown price source: ${source.name}`)
    }
  }

  async fetchFromOraclesCash(source) {
    const controller = new AbortController()
    const timeoutId = setTimeout(() => controller.abort(), 10000)

    try {
      const response = await fetch(source.url, {
        method: 'GET',
        headers: {
          Accept: 'application/json',
          'Content-Type': 'application/json',
        },
        signal: controller.signal,
      })

      if (!response.ok) {
        throw new Error(`Oracle API returned status: ${response.status}`)
      }

      const data = await response.json()
      const usdBchOracle = data.oracles.find((oracle) => oracle.publicKey === ORACLE_PUBKEY)

      if (!usdBchOracle || !usdBchOracle.messageMetrics) {
        throw new Error('USD/BCH oracle not found or has no message metrics')
      }

      const currentPriceInCents = usdBchOracle.messageMetrics.currentPrice
      const priceInUSD = currentPriceInCents / 100

      // Get oracle message for signature
      const messageResponse = await fetch(
        `https://oracles.generalprotocols.com/api/v1/oracleMessages?publicKey=${ORACLE_PUBKEY}&count=1`
      )
      
      const messageData = await messageResponse.json()
      const oracleMessage = messageData.oracleMessages[0]

      clearTimeout(timeoutId)

      return {
        price: priceInUSD,
        message_hex: oracleMessage.message,
        signature_hex: oracleMessage.signature,
        oracle_pubkey_hex: ORACLE_PUBKEY,
        source: 'oracles.cash'
      }

    } catch (error) {
      clearTimeout(timeoutId)
      throw error
    }
  }

  async fetchFromCoinGecko(source) {
    const controller = new AbortController()
    const timeoutId = setTimeout(() => controller.abort(), 8000)

    try {
      const response = await fetch(`${source.url}?ids=bitcoin-cash&vs_currencies=usd&include_market_cap=false`, {
        method: 'GET',
        headers: {
          Accept: 'application/json',
        },
        signal: controller.signal,
      })

      if (!response.ok) {
        throw new Error(`CoinGecko API returned status: ${response.status}`)
      }

      const data = await response.json()
      const price = data['bitcoin-cash']?.usd

      if (!price || typeof price !== 'number') {
        throw new Error('Invalid price data from CoinGecko')
      }

      clearTimeout(timeoutId)

      return {
        price: price,
        source: 'coingecko'
      }

    } catch (error) {
      clearTimeout(timeoutId)
      throw error
    }
  }

  async fetchFromCoinMarketCap(source) {
    const controller = new AbortController()
    const timeoutId = setTimeout(() => controller.abort(), 8000)

    try {
      const response = await fetch(`${source.url}?symbol=BCH&convert=USD`, {
        method: 'GET',
        headers: {
          'X-CMC_PRO_API_KEY': process.env.VUE_APP_COINMARKETCAP_API_KEY,
          Accept: 'application/json',
        },
        signal: controller.signal,
      })

      if (!response.ok) {
        throw new Error(`CoinMarketCap API returned status: ${response.status}`)
      }

      const data = await response.json()
      const price = data.data?.BCH?.quote?.USD?.price

      if (!price || typeof price !== 'number') {
        throw new Error('Invalid price data from CoinMarketCap')
      }

      clearTimeout(timeoutId)

      return {
        price: price,
        source: 'coinmarketcap'
      }

    } catch (error) {
      clearTimeout(timeoutId)
      throw error
    }
  }

  validatePrice(priceData) {
    if (!priceData || !priceData.price) {
      return false
    }

    const price = parseFloat(priceData.price)
    
    // Price range validation
    if (price < 50 || price > 10000) {
      console.warn(`Price ${price} out of reasonable range`)
      return false
    }

    return true
  }

  calculateConsensusPrice(priceData) {
    if (priceData.length === 1) {
      return priceData[0]
    }

    // Calculate average price
    const prices = priceData.map(d => d.price)
    const average = prices.reduce((sum, price) => sum + price, 0) / prices.length
    
    // Check for significant deviations
    const maxDeviation = 0.05 // 5% max deviation
    const consensusPrices = priceData.filter(d => 
      Math.abs(d.price - average) / average <= maxDeviation
    )

    if (consensusPrices.length === 0) {
      return null // No consensus
    }

    // Return the most reliable source from consensus prices
    return consensusPrices.sort((a, b) => {
      const healthA = this.getSourceHealth(a.source)
      const healthB = this.getSourceHealth(b.source)
      return healthA.priority - healthB.priority
    })[0]
  }

  getFallbackPrice(error) {
    return {
      price: 350.0, // Conservative fallback price
      message_hex: '0000000000007530',
      signature_hex: '3044022044f7d0438553f6fb52be62a94e6d676c6d47536f6a101f51f76257931db14030022009073ba73e721684db25397e73d6c210',
      oracle_pubkey_hex: ORACLE_PUBKEY,
      source: 'fallback',
      status: 'fallback',
      error: error?.message
    }
  }

  updateSourceHealth(sourceName, status) {
    this.sourceHealth.set(sourceName, {
      status: status,
      lastCheck: Date.now(),
      priority: this.priceSources.find(s => s.name === sourceName)?.priority || 999
    })
  }

  getSourceHealth(sourceName) {
    return this.sourceHealth.get(sourceName) || { status: 'unknown', priority: 999 }
  }

  clearCache() {
    this.priceCache.clear()
  }
}

// Usage example

// Fetch price with fallbacks
const priceData = await oracleService.fetchPriceWithFallbacks()
console.log('Price data:', priceData)
