// Enhanced analytics service with performance tracking
export class AnalyticsService {
  constructor() {
    this.metrics = new Map()
    this.sessionId = this.generateSessionId()
    this.startTime = Date.now()
    this.performanceThresholds = {
      apiResponseTime: 2000, // 2 seconds
      transactionTime: 30000, // 30 seconds
      renderTime: 100 // 100ms
    }
  }

  generateSessionId() {
    return 'session_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9)
  }

  // Enhanced vault operation tracking
  trackVaultOperation(operation, vaultId, metadata = {}) {
    const event = {
      event: `vault_${operation}`,
      vaultId: vaultId,
      timestamp: Date.now(),
      sessionId: this.sessionId,
      metadata: {
        ...metadata,
        userAgent: navigator.userAgent,
        screenResolution: `${screen.width}x${screen.height}`,
        connectionType: this.getConnectionType()
      }
    }

    // Track performance metrics
    this.trackPerformance(operation, event)

    // Send to analytics backend
    this.sendEvent(event)

    // Store locally for real-time dashboard
    this.storeEvent(event)
  }

  trackPerformance(operation, event) {
    const performanceData = {
      operation: operation,
      timestamp: Date.now(),
      sessionId: this.sessionId,
      metrics: {
        memoryUsage: this.getMemoryUsage(),
        cpuUsage: this.getCPUUsage(),
        networkLatency: this.getNetworkLatency(),
        renderTime: this.getRenderTime()
      }
    }

    this.metrics.set(`performance_${operation}_${Date.now()}`, performanceData)

    // Check for performance issues
    this.checkPerformanceThresholds(performanceData)
  }

  // Enhanced API call tracking
  trackApiCall(apiUrl, method, duration, success, error = null) {
    const apiEvent = {
      event: 'api_call',
      apiUrl: apiUrl,
      method: method,
      duration: duration,
      success: success,
      error: error,
      timestamp: Date.now(),
      sessionId: this.sessionId,
      metadata: {
        userAgent: navigator.userAgent,
        onlineStatus: navigator.onLine
      }
    }

    // Check API performance thresholds
    if (duration > this.performanceThresholds.apiResponseTime) {
      this.reportPerformanceIssue('api_slow_response', apiEvent)
    }

    this.sendEvent(apiEvent)
    this.storeEvent(apiEvent)
  }

  // Transaction tracking with performance metrics
  trackTransaction(vaultId, transactionType, duration, success, txId = null, error = null) {
    const txEvent = {
      event: 'transaction',
      vaultId: vaultId,
      transactionType: transactionType,
      duration: duration,
      success: success,
      txId: txId,
      error: error,
      timestamp: Date.now(),
      sessionId: this.sessionId,
      metadata: {
        networkType: this.getNetworkType(),
        walletType: this.getWalletType(),
        gasUsed: error ? null : this.getGasUsed(txId)
      }
    }

    // Check transaction performance
    if (duration > this.performanceThresholds.transactionTime) {
      this.reportPerformanceIssue('transaction_slow', txEvent)
    }

    this.sendEvent(txEvent)
    this.storeEvent(txEvent)
  }

  // Error tracking with context
  trackError(error, context = {}) {
    const errorEvent = {
      event: 'error',
      error: {
        message: error.message,
        stack: error.stack,
        name: error.name
      },
      context: {
        ...context,
        sessionId: this.sessionId,
        timestamp: Date.now(),
        userAgent: navigator.userAgent,
        url: window.location.href,
        userId: this.getUserId()
      }
    }

    this.sendEvent(errorEvent)
    this.storeEvent(errorEvent)

    // Categorize error for better analysis
    this.categorizeError(errorEvent)
  }

  // Performance monitoring utilities
  getMemoryUsage() {
    if (performance.memory) {
      return {
        used: Math.round(performance.memory.usedJSHeapSize / 1048576), // MB
        total: Math.round(performance.memory.totalJSHeapSize / 1048576), // MB
        limit: Math.round(performance.memory.jsHeapSizeLimit / 1048576) // MB
      }
    }
    return null
  }

  getCPUUsage() {
    // Simplified CPU usage estimation
    const start = performance.now()
    const end = performance.now()
    return Math.round((end - start) * 100) / 100
  }

  getNetworkLatency() {
    if ('connection' in navigator) {
      return {
        effectiveType: navigator.connection.effectiveType,
        downlink: navigator.connection.downlink,
        rtt: navigator.connection.rtt
      }
    }
    return null
  }

  getRenderTime() {
    if (performance.getEntriesByType) {
      const paintEntries = performance.getEntriesByType('paint')
      return paintEntries.map(entry => ({
        name: entry.name,
        startTime: Math.round(entry.startTime),
        duration: Math.round(entry.duration)
      }))
    }
    return null
  }

  getConnectionType() {
    if ('connection' in navigator) {
      return navigator.connection.effectiveType
    }
    return 'unknown'
  }

  getNetworkType() {
    // Determine if testnet or mainnet based on configuration
    return process.env.VUE_APP_NETWORK_TYPE || 'chipnet'
  }

  getWalletType() {
    // Get wallet type from Vuex store
    return this.$store?.state?.wallet?.type || 'unknown'
  }

  getGasUsed(txId) {
    // This would typically come from blockchain API
    return null // Placeholder
  }

  getUserId() {
    // Get user ID from Vuex store or local storage
    return this.$store?.state?.wallet?.address || 'anonymous'
  }

  // Performance threshold checking
  checkPerformanceThresholds(performanceData) {
    const { metrics } = performanceData

    if (metrics.memoryUsage) {
      const memoryUsagePercent = (metrics.memoryUsage.used / metrics.memoryUsage.total) * 100
      if (memoryUsagePercent > 80) {
        this.reportPerformanceIssue('high_memory_usage', performanceData)
      }
    }

    if (metrics.renderTime) {
      const slowRenders = metrics.renderTime.filter(render => render.duration > this.performanceThresholds.renderTime)
      if (slowRenders.length > 0) {
        this.reportPerformanceIssue('slow_render', performanceData)
      }
    }
  }

  reportPerformanceIssue(issueType, data) {
    const issue = {
      type: issueType,
      data: data,
      timestamp: Date.now(),
      sessionId: this.sessionId,
      severity: this.getIssueSeverity(issueType, data)
    }

    console.warn(`Performance issue detected: ${issueType}`, issue)

    // Send performance issue to monitoring
    this.sendEvent({
      event: 'performance_issue',
      issue: issue
    })
  }

  getIssueSeverity(issueType, data) {
    switch (issueType) {
      case 'high_memory_usage':
        return data.metrics.memoryUsage.used / data.metrics.memoryUsage.total > 0.9 ? 'critical' : 'warning'
      case 'api_slow_response':
        return data.duration > 5000 ? 'critical' : 'warning'
      case 'transaction_slow':
        return data.duration > 60000 ? 'critical' : 'warning'
      case 'slow_render':
        return 'warning'
      default:
        return 'info'
    }
  }

  categorizeError(errorEvent) {
    const { error, context } = errorEvent

    let category = 'unknown'
    if (error.message.includes('network') || error.message.includes('fetch')) {
      category = 'network'
    } else if (error.message.includes('wallet') || error.message.includes('sign')) {
      category = 'wallet'
    } else if (error.message.includes('oracle') || error.message.includes('price')) {
      category = 'oracle'
    } else if (error.message.includes('transaction') || error.message.includes('contract')) {
      category = 'blockchain'
    }

    errorEvent.category = category
  }

  // Event storage and transmission
  storeEvent(event) {
    // Store in localStorage for real-time dashboard
    const key = `analytics_events_${this.sessionId}`
    const existingEvents = JSON.parse(localStorage.getItem(key) || '[]')
    existingEvents.push(event)
    
    // Keep only last 100 events to prevent storage overflow
    if (existingEvents.length > 100) {
      existingEvents.splice(0, existingEvents.length - 100)
    }
    
    localStorage.setItem(key, JSON.stringify(existingEvents))
  }

  async sendEvent(event) {
    try {
      // Send to analytics backend
      const response = await fetch('/api/analytics', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify(event)
      })

      if (!response.ok) {
        throw new Error(`Analytics API error: ${response.status}`)
      }
    } catch (error) {
      console.error('Failed to send analytics event:', error)
      // Store failed events for retry
      this.storeFailedEvent(event)
    }
  }

  storeFailedEvent(event) {
    const key = 'failed_analytics_events'
    const failedEvents = JSON.parse(localStorage.getItem(key) || '[]')
    failedEvents.push(event)
    localStorage.setItem(key, JSON.stringify(failedEvents))
  }

  // Retry failed events
  async retryFailedEvents() {
    const key = 'failed_analytics_events'
    const failedEvents = JSON.parse(localStorage.getItem(key) || '[]')
    
    if (failedEvents.length === 0) {
      return
    }

    const retryPromises = failedEvents.map(event => this.sendEvent(event))
    
    try {
      await Promise.allSettled(retryPromises)
      // Clear failed events on successful retry
      localStorage.removeItem(key)
    } catch (error) {
      console.error('Some failed events could not be retried:', error)
    }
  }

  // Get analytics data for dashboard
  getAnalyticsData() {
    const key = `analytics_events_${this.sessionId}`
    const events = JSON.parse(localStorage.getItem(key) || '[]')
    
    return {
      events: events,
      sessionInfo: {
        sessionId: this.sessionId,
        startTime: this.startTime,
        duration: Date.now() - this.startTime,
        totalEvents: events.length
      },
      performance: this.getPerformanceSummary()
    }
  }

  getPerformanceSummary() {
    const performanceEvents = Array.from(this.metrics.values())
      .filter(event => event.metrics)
    
    if (performanceEvents.length === 0) {
      return null
    }

    const avgMemory = performanceEvents.reduce((sum, event) => {
      return sum + (event.metrics.memoryUsage?.used || 0)
    }, 0) / performanceEvents.length

    const avgLatency = performanceEvents.reduce((sum, event) => {
      return sum + (event.metrics.networkLatency?.rtt || 0)
    }, 0) / performanceEvents.length

    return {
      averageMemoryUsage: Math.round(avgMemory),
      averageNetworkLatency: Math.round(avgLatency),
      totalPerformanceEvents: performanceEvents.length
    }
  }
}
